# 进阶数据结构专题

# 平衡二叉树AVL

名称：AVL树

发明者：

​ G.M. Adelson-Velsky & E.M. Landis

年代： 1962年（55岁）

性质：

$|H(left) - H(right)| <= 1$

优点：

由于对每个节点的左右子树的树高做了限制，所以整棵树不会退化成一个链表

问题：

  1. 高度为H的BS树，所含的节点数量在什么范围内？

    $H <= SIZE(h) <= 2^n - 1$

  2. 高度为H的AVL树，所包含的节点数量在什么范围之内？

    **low(h)**表示高度为h，AVL树所包含的最少节点的数量。

    所以 $2^h - 1 >= size(h) >= low(h - 1) + low(h - 2) + 1$

    因为左右子树差别为1，所以包含的节点数下限为low(h - 1)(高一层子树的最少节点)，加上 high(h - 2)(矮一层子树的最少节点)， 加上 1 (一个根节点)

    $low(1) = 1, low(2) = 2, low(3) = 4, ...$ 从3开始就可以往后递推了（斐波那契数列）增长趋势为 $1.618^n \approx 1.5^n$

所以 AVL树提高了树形结构查找下限，保证logN。

AVL树操作：

  • 左旋

    1. 左旋将原先根节点的右孩子置为根节点
    2. 将原先的根节点置为左孩子
    3. 将原先右孩子的左子树置为原先根节点的右子树（因为它上的所有值都比这个原先的根节点都大

  • 右旋：左旋的逆操作

    1. 将原先根节点的左孩子置为根节点
    2. 将原先根节点置为右孩子
    3. 将原先左孩子的右子树置为新的右孩子的左子树

# AVL树-四种失衡类型

LL型： 左孩子的左子树树高高一点

RR型：右孩子的右子树树高高一点

LR型： 左孩子的右子树树高高一点

RL型： 右孩子的左子树树高高一点

**对于LL型：**右旋

$$ hk_1 = hk_2 + 1 \ hk_2 = hk_3 + 2 \ h_A = h_B + 1 \ hk_3 = max(h_C, h_D) + 1 \ h_A = max(h_C, h_D) + 2 $$

**对于RR型：**是LL型镜像，左旋即可。

对于LR型：

$$h_{k_1} = h_{k_2} + 1 \ h_{k_2} = h_D + 2 \ h_{k_2} = h_{k_3} + 1 \ h_{k_3} = h_A + 1 \ h_{k_3} = max(h_B, h_C) + 1 \ h_A + 1 = max(h_B, h_C) + 1 \ \therefore h_A = max(h_B, h_C) = h_D$$

  1. LR->先左旋左子树->LL

    2.LL->再右旋整个树

**对于RL型：**是LR的镜像，先右旋右子树，再大左旋

#include <iostream>
#include <cstdio>
#include <cstdlib>
using namespace std;

struct Node {
    int key, h;
    Node *lchild, *rchild;
    // Node() : key(0), h(0), lchild(nullptr), rchild(nullptr) {};
    // Node(int _key) : key(_key), h(1), lchild(nullptr), rchild(nullptr) {}
};

// decrease the judge cases, we can do like the following:
// use NIL to replace all the nullptr
Node __NIL;
#define NIL (&__NIL)
__attribute__((constructor)) // make NIL extand in advance
void init_NIL() {
    NIL->key = 0, NIL->h = 0; 
    NIL->lchild = NIL->rchild = NIL;
    return ;
}

inline int H(Node *root) {
    return root != NIL ? root->h : 0;
}

inline int KEY(Node *root) {
    return root != NIL? root->key : 0;
}

Node *getNewNode(int val) {
    Node *p = (Node *)malloc(sizeof(Node));
    p->key = val;
    p->h = 1;
    p->lchild = p->rchild = NIL;
    return p;
}

// when we insert a node, we need to update the height
void update_height(Node *root) {
    int h1 = H(root->lchild);
    int h2 = H(root->rchild);
    root->h = (h1 > h2 ? h1 : h2) + 1;
    return ;
}

// right rotate
Node *right_rotate(Node *root) {
    Node *tmp = root->lchild;
    root->lchild = tmp->rchild; // 
    tmp->rchild = root;
    update_height(root); // root as child
    update_height(tmp); // tmp as new root
    return tmp; // be marked as root
}

// left rotate 
Node *left_rotate(Node *root) {
    Node *tmp = root->rchild;
    root->rchild = tmp->lchild;
    tmp->lchild = root;
    update_height(root); // root as left child
    update_height(tmp); // tmp as new root
    return tmp;
}

            root->lchild = left_rotate(root->lchild);
        }
        root = right_rotate(root); // LL or LR
    } else {
        if (root->rchild->lchild->h > root->rchild->rchild->h) { // RL
            root->rchild = right_rotate(root->rchild);
        }
        root = left_rotate(root); // RR or RL
    }
    return root;
}

Node *insert(Node *root, int val) {
    if (root == NIL) return getNewNode(val);
    if (root->key == val) return root;
    else if (root->key > val) {
        root->lchild = insert(root->lchild, val);
    } else {
        root->rchild = insert(root->rchild, val);
    }
    update_height(root);
    return maintain(root);
}

// the root has two out-degree
Node *predecedure(Node *root) {
    Node *l = root->lchild;
    while (l->rchild != NIL) {
        l = l->rchild;
    }
    return l;
}

Node *erase(Node *root, int val) {
    if (root == NIL) return NIL;
    if (root->key > val) {
        root->lchild = erase(root->lchild, val);
    } else if (root->key < val) {
        root->rchild = erase(root->rchild, val);
    } else if (root->key == val) {
        if (root->lchild == NIL && root->rchild == NIL) {
            delete root;
            root = NIL;
        } else if (root->lchild == NIL || root->rchild == NIL) {
            Node *tmp = (root->lchild != NIL ? root->lchild : root->rchild);
            delete root;
            root = NIL;
            return tmp;
        } else {
            Node *pre = predecedure(root);
            root->key = pre->key;
            root->lchild = erase(root->lchild, pre->key);
        }
    }
    return root; // the hidden case is that if val was not found, nothing would be changed.
}

void clear(Node *root) {
    if (root == NIL) return ;
    clear(root->lchild);
    clear(root->rchild);
    delete root;
    root = NIL;
    return ;
}

void output(Node *root) {
    if (root == NIL) return ;
    output(root->lchild);
    cout << root->key << " -> ";
    output(root->rchild);
    return ;
}

void outS(Node *root) {
    if (root == NIL) return ;
    cout << KEY(root) << "[" << root->h << "]" << "(" << KEY(root->lchild) << ", " << KEY(root->rchild) << ")" << endl;
    outS(root->lchild);
    outS(root->rchild);
    return ;
}

int main() {
    int op, val;
    Node *root = NIL;
    while (cin >> op >> val) {
        switch (op) {
            case 1: root = insert(root, val); break;
            case 2: root = erase(root, val); break;
        }
        output(root);
        cout << endl;
        outS(root);
        cout << endl << "-----------------------------" << endl;
    }
    return 0;
}

# SB树（Size Balanced 树）

性质： $$ Size(left) >= Size(right ->left) \ Size(left) >= Size(right ->right) \ Size(right) >= Size(left ->left) \ Size(right) >= Size(left -> right) $$ **优点：**由于对每个节点的左右子树的节点数量做了限制，所以整棵树不会退化成一个链表
