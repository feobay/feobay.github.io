#include <iostream>
using namespace std;

const int N = 1e5 + 10;
int arr[N], n;

void quickSort(int *arr, int l, int r) {
    if (l >= r) return;
    int x = arr[(l + r) >> 1];
    int i = l - 1, j = r + 1;
    while (i < j) {
        do i++; while (arr[i] < x);
        do j--; while (arr[j] > x);
        if (i < j) swap(arr[i], arr[j]);
    }
    quickSort(arr, l, j), quickSort(arr, j + 1, r);
}

int main() {
    cin >> n;
    for (int i = 0; i < n; ++i) cin >> arr[i];
    quickSort(arr, 0, n - 1);
    for (int i = 0; i < n; ++i) cout << arr[i] << ' ';
    return 0;
}

#include <iostream>

const int N = 1e5 + 10;
int n, a[N], tmp[N];
long long res; // 可能爆int

void getRe(int *arr, int l, int r) {
    if (l == r) return ;
    int mid = (l + r) >> 1;
    getRe(arr, l, mid), getRe(arr, mid + 1, r);
    int i = l, j = mid + 1, k = 0;
    while (i <= mid && j <= r) {
        if (arr[i] <= arr[j]) {
            tmp[k++] = arr[i++];
        }
        else {
            res += mid - i + 1;
            tmp[k++] = arr[j++];
        }

    while (j <= r) tmp[k++] = arr[j++];
    for (i = l, k = 0; i <= r; ++i, ++k) arr[i] = tmp[k];
}

ACW.791高精度加法

#include <iostream>
#include <string>
#include <algorithm>
using namespace std;

string addInteger(string a, string b) {
    if (a.size() < b.size()) swap(a, b); // 长整数在前
    reverse(a.begin(), a.end());
    reverse(b.begin(), b.end());
    string ans;
    int c = 0, tmp; // c 为进位，tmp为当前位的和
    int i;
    for (i = 0; i < a.size() && i < b.size(); ++i) {
        tmp = a[i] - '0' + b[i] - '0' + c;
        ans += tmp % 10 + '0';
        c = tmp / 10;
    }
    // 继续补完a剩余的长度
    for (; i < a.size(); ++i) {
        tmp = a[i] - '0' + c;
        ans += tmp % 10 + '0';
        c = tmp / 10;
    }
    if (c) ans += c + '0';
    reverse(ans.begin(), ans.end());
    return ans;
}

int main() {
    string a, b;
    cin >> a >> b;
    cout << addInteger(a, b) << endl;
    return 0;
}

# 1.DFS(Deep First Search)

例题，输出一组数的全排列。

#include <iostream>
#define IOS {ios::sync_with_stdio(false), cin.tie(0), cout.tie(0);}
using namespace std;

const int N = 10;

int n;
int path[N];
bool st[N]; //state

void dfs(int u) { //u表示当前遍历层数
    if (u == n) { //当层数u为最深层数n时，输出路径，并返回
        for (int i  = 0; i < n; i++) {
            cout << path[i] << ' ';
        }
        puts("");
        return;
    }

    for (int i = 1; i <= n; i++) {
        if (!st[i]) {
            path[u] = i;
            st[i] = true;
            dfs(u + 1);
            st[i] = false; //每置一次false就相当于向上返回一层
            path[u] = 0; //与上一句一起用于恢复上一层的现场
        }
    }
}

int main() {
    IOS
    cin >> n;
    dfs(0);
    return 0;
}

# 2.BFS(Breadth First Search)

例题，走迷宫，宽搜：

当所有边的权重相同时，才可以用宽搜来做最短路。

#include <iostream>
#include <queue>
using namespace std;

struct node {
    int x, y, d;  
};

int n, m;
const int N = 105;
int a[N][N];
bool st[N][N];

int movex[] = {0, 0, -1, 1}, movey[] = {1, -1, 0, 0};

int bfs() {
    queue<node> q;
    q.push({1, 1, 0});
    st[1][1] = true;
    while (q.size()) {
        node t = q.front();
        q.pop();
        for (int i = 0; i < 4; ++i) {
            int nx = t.x + movex[i], ny = t.y + movey[i];
            if (nx < 1 || nx > n || ny < 1 || ny > m || st[nx][ny] || a[nx][ny]) continue;
            if (nx == n && ny == m) return t.d + 1;
            st[nx][ny] = true;
            q.push({nx, ny, t.d + 1});
        }
    }
}

int main() {
    scanf("%d%d", &n, &m);
    for (int i = 1; i <= n; ++i) {
        for (int j = 1; j <= m; ++j) {
            scanf("%d", &a[i][j]);
        }
    }
    printf("%d\n", bfs());
    return 0;
}

# 3.树和图的存储

树是一种无环图，所以按照图来存储即可

图分为有向图和无向图。无向图可去可会，意味着相当于有向图在两个点之间建立一去一回两条边，所以只用考虑有向图如何存储即可。

#include <iostream>
#include <cstring>
#include <algorithm>
using namespace std;

const int N = 1e5 + 10, M = N * 2; // N
int n, m;
int h[N], e[M], ne[M], idx; // h为头，e为节点的值，ne为节点的下一个节点的索引，idx为节点的索引
bool st[N];

int ans = N;

void add(int a, int b) { // 在连通块a中插入节点b
    e[idx] = b, ne[idx] = h[a], h[a] = idx++;
}

int dfs(int u) {
    st[u] = true; //标记以下，已经被搜过了

    int sum = 1, res = 0;
    for (int i = h[u]; i != -1; i = ne[i]) {
        int j = e[i];
        if (!st[j]) {
            int s = dfs(j);
            res = max(res, s);
            sum += s;
        }
    }

    res = max(res, n - sum);
    ans = min(ans, res);
    return sum;
}

int main() {
    memset(h, -1, sizeof(h));
    cin >> n;
    for (int i = 0; i < n - 1; ++i) {
        int a, b;
        cin >> a >> b;
        add(a, b), add(b, a);
    }

    dfs(1);
    cout << ans << endl;
    return 0;
}

# 5.朴素Dijkstra

步骤：

1.初始化各边（g[x][y]）大小，存在则更新为该大小，不存在则更新为正无穷0x3f3f3f3f; 初始化各点到起点的路径距离，除了起点为0（dist[1] = 0), 其余均为正无穷0x3f3f3f3f(memset(g, 0x3f, sizeof g))

2.进行n次迭代，每次迭代都换一点作为更新的点

​ 2.1找到当前距离起点最小的点t

​ 2.2使用t来更新其它点到起点的距离dist[j]

e.g ACWING.Dijkstra求最短路 I (opens new window)

#include <iostream>
#include <algorithm>
#include <string>
#include <cstring>
using namespace std;

const int N = 510;

int n, m;
int g[N][N]; //存储每个边的距离（邻接表）
int dist[N]; //存储每个点到起点1的距离
bool st[N]; //已遍历点集合

// o(n^2)
int dijkstra() {
    // 1.初始化
    memset(dist, 0x3f, sizeof dist);
    dist[1] = 0;

    // 2.迭代找出最小点
    for(int j = 1; j <= n; j++) { //n个点，m条边
        int t = -1; //t用来存储当前未更新完的点中到起点距离最小的点

        //2.1 遍历所有点找出当前未被遍历的最小点，记录进t
        for (int j = 1; j <= n; j++) {
            if (!st[j] && (t == -1 || dist[t] > dist[j])) t = j;
        }
        st[t] = true; //找到最小的点并以它作为起点更新其出度指向点
        //作为下一次的起点则将该点放入已遍历集合

        //2.2 用当前点作为起点更新所有点到1的距离（实际上就是遍历所有点）
        for (int j = 1; j <= n; j++) {
            dist[j] = min(dist[j], dist[t] + g[t][j]); //不用判断，不存在的边是正无穷
        }
    }
    if (dist[n] == 0x3f3f3f3f) return -1; //说明1与n不连通
    return dist[n]; //否则返回最短距离 
}

int main() {
    scanf("%d%d", &n, &m);
    memset(g, 0x3f, sizeof g);

    while (m -- ) {
        int a, b, c;
        scanf("%d%d%d", &a, &b, &c);
        g[a][b] = min(g[a][b], c); //有可能在输入阶段重复输入同一条边，但长度不同
    }
    int t = dijkstra();
    printf("%d\n", t);
    return 0;
}

# 6.堆优化版Dijkstra

改用邻接表存储，每次将已经确定长度的点放入堆中

Dijkstra II (opens new window)

#include <iostream>
#include <cstring>
#include <queue>
using namespace std;

using PII = pair<int, int>;

const int N = 150010;

//稀疏图使用邻接表存储
int idx, e[N], ne[N], h[N]; //idx相当于指针，e存的值为idx指向边的入度点（终点），ne为idx的next指针，h[x]为以x为起点的边的指针
int w[N]; //存权重,即为每个出边的大小,每条边都有个独特的地址idx来存
int dist[N];//存每个点到起点的距离
bool st[N];//存每个点是否被访问

int n, m;

void add(int x, int y, int c) {
    w[idx] = c; //重边不要紧，会更新两次放入堆中，最终的结果仍然是按照最小的来计算
    e[idx] = y;
    ne[idx] = h[x];
    h[x] = idx++;
}

int dijkstra() {
    memset(dist, 0x3f, sizeof dist);
    dist[1] = 0;
    priority_queue<PII, vector<PII>, greater<PII>> heap; //小根堆
    heap.push({0, 1}); //堆根据pair的第一个值排序，所以第一个值为距离，第二个值为对应的点索引 
    while(heap.size()) {
        PII k = heap.top();
        heap.pop();
        int ind = k.second, dis = k.first;
        
        if (st[ind]) continue;
        st[ind] = true;

        for (int i = h[ind]; i != -1; i = ne[i]) {
            int j = e[i];
            if (dist[j] > dis + w[i]) {
                dist[j] = dis + w[i];
                heap.push({dist[j], j});
            }
        }
    }
    if (dist[n] == 0x3f3f3f3f) return -1;
    return dist[n];
}

int main() {
    memset(h, -1, sizeof h);
    cin >> n >> m;
    while (m--) {
        int x, y, z;
        cin >> x >> y >> z;
        add(x, y, z);
    }

    cout << dijkstra() << endl;
    return 0;
}

# 7.Bellman-Ford算法

贝尔曼算法

若有负权回路，且负权在某一条到终点的路径上，那么就会使dist变为负无穷，那么就不存在最短路。

或者说限制路径最多边数，负权回路也可以求出最短路（此时就需要使用Bellman算法）

Bellman-Ford算法注意点：

**串联：**同一个路径上在一次遍历中就前进了两条边（就无法通过外层循环限制边数了）

// 1 ~ n 的数字中，x出现的次数
int countBit(int n, int x) {
    if (!n) return 0;
    // 将n的每一位从低位开始依次放入vector中
    vector<int> num;
    while (n) {
        num.push_back(n % 10);
        n /= 10;
    }
    n = num.size();

    int res = 0;
    // 从高位开始算起，x为0则不从上线高位开始算
    for (int i = n - 1 - !x; i >= 0; --i) {
        if (i < n - 1) { // x 在 num[i]这个位置出现的次数，num[i] 前面部分的数在[1, get(num, i + 1, n - 1) - 1]之间的情况
            res += get(num, i + 1, n - 1) * power10(i);
            if(!x) res -= power10(i); // 减去为0的情况
        }
        // 前面的数等于上限的情况
        if (num[i] == x) res += get(num, 0, i - 1) + 1; // 加上末尾为0的情况
        else if (num[i] > x) res += power10(i);
    }
    return res;
}

int main() {
    int a, b;
    while (cin >> a >> b, a || b) {
        if (a > b) swap(a, b);

        for (int i = 0; i < 10; ++i) {
            cout << countBit(b, i) - countBit(a - 1, i) << ' ';
        }
        cout << endl;
    }
    return 0;
}

ACW905.区间选点

给定 $N$ 个闭区间 $[a_i,b_i]$，请你在数轴上选择尽量少的点，使得每个区间内至少包含一个选出的点。

输出选择的点的最小数量。

位于区间端点上的点也算作区间内。

输入格式

第一行包含整数 $N$，表示区间数。

接下来 $N$ 行，每行包含两个整数 $a_i,b_i$表示一个区间的两个端点。

输出格式

输出一个整数，表示所需的点的最小数量。

数据范围

$1, ≤ , N , ≤ , 10^5$ $−10^9 , ≤ , a_i , ≤ , b_i , ≤ ,10^9$

输入样例：

3
-1 1
2 4
3 5

输出样例：

2

解法：按照右端点小到大排序

#include <iostream>
#include <algorithm>
using namespace std;

// 区间选点
// 1.将每个区间按右端点从小到大排序
// 2.从前往后依次枚举每个区间，如果当前区间中已经包含点，则直接pass
//   否则，选择当前区间的右端点

const int N = 1e5 + 10;

struct Range {
    int l, r;
    bool operator<(const Range &w) const {
        return r < w.r;
    }
} range[N];


int main() {
    int n;
    scanf("%d", &n);
    int l, r;
    for (int i = 0; i < n; ++i) {
        scanf("%d%d", &l, &r);
        range[i] = {l, r};
    }

    sort(range, range + n);
    int ans = 0, ed = -2e9;
    for (int i = 0; i < n; ++i) {
        if (range[i].l > ed) {
            ++ans;
            ed = range[i].r;
        }
    }
    printf("%d", ans);
    return 0;
}

/*
证明：

假设最优解为 ans 个不相交区间，以上述贪心思路选出来的区间为 cnt 个。即证明 ans = cnt，等价于 ans >= cnt && ans <= cnt
首先，以上述贪心思路选择出的 cnt 个区间，是一组可行方案。其覆盖了所有区间，满足题目要求。且 ans 表示的是所有可行方案的最大值，那么 ans >= cnt 成立
证明 ans <= cnt 时，可以采用反证法。假设 ans > cnt，则说明最多有 ans 个两两不交的区间，则至少需要 ans 个点才能将这些区间全部覆盖，然而实际上只需要 cnt 个点就能覆盖全部区间，且 cnt < ans。则矛盾，故 ans <= cnt 成立
时间复杂度：O ( n l o g n ) O(nlogn)O(nlogn)

ACW906.区间分组

#include <iostream>
#include <algorithm>
#include <queue>
using namespace std;

// 做法
// 1.将区间按照左端点进行排序
// 2.从前往后处理每个组的区间
//      判断能否将其放到某个现有的组中 L[i] > Max_r
//          1.如果不存在这样的组，则开新组，然后再将其放进去；
//          2.如果存在这样的组，将其放进去，并更新当前组的Max_r

const int N = 1e5 + 10;
struct Range {
    int l, r;
    bool operator<(const Range &w) const {
        return l < w.l;
    }
} range[N];

int main() {
    int n;
    cin >> n;
    int l, r;
    for (int i = 0; i < n; ++i) {
        cin >> l >> r;
        range[i] = {l, r};
    }
    sort(range, range + n);
    priority_queue<int, vector<int>, greater<int>> heap; // 小根堆，最小的边界在最上方
    for (int i = 0; i < n; ++i) {
        auto tmp = range[i];
        if (heap.empty() || heap.top() >= tmp.l) heap.push(tmp.r); // 有交集或者堆为空，则将该边界放入heap，相当于新建组
        else {
            // int t = heap.top();
            heap.pop(); // pop出同组的右边界
            heap.push(tmp.r); // 更新右边界
        }
    }
    cout << heap.size() << endl;
    return 0;
}

ACW907.区间覆盖

#include <iostream>
#include <algorithm>
using namespace std;

// 1.将所有区间按左端点从小到大排序
// 2.从前往后依次枚举每个区间，在所有能覆盖的start的区间中，选择右端点最大的区间
//   然后将start更新成右端点的最大值

const int N = 1e5 + 10;

int n;
struct Range {
    int l, r;
    bool operator<(const Range &w) const {
        return l < w.l;
    }
} range[N];

int main() {
    int st, ed;
    cin >> st >> ed;
    cin >> n;
    int l, r;
    for (int i = 0; i < n; ++i) {
        cin >> l >> r;
        range[i] = {l, r};
    }
    sort(range, range + n);
    int ans = 0;
    bool success = false;
    for (int i = 0; i < n; ++i) {
        int j = i, r = -2e9;
        // 寻找所有 左端点在st左边的区间 的 右端点的最大值
        while (j < n && range[j].l <= st) {
            r = max(r, range[j].r);
            ++j;
        }
        if (r < st) {
            ans = -1;
            break;
        }
        ++ans;
        if (r >= ed) {
            success = true;
            break;
        }
        st = r;
        i = j - 1; // 最后计算到的区间，因为后面还要++i，所以是j - 1
    }
    if (!success) cout << -1 << endl;
    else cout << ans << endl;
    return 0;
}

ACW148.合并果子

每次选择最小的两个数

#include <iostream>
#include <queue>
using namespace std;

// 每次选最小的两个合并(哈夫曼树)

int main() {
    int n;
    cin >> n;
    priority_queue<int, vector<int>, greater<int>> heap;
    int x;
    while (n--) {
        cin >> x;
        heap.push(x);
    }
    int a, b, res = 0;
    while (heap.size() > 1) {
        a = heap.top(); heap.pop();
        b = heap.top(); heap.pop();
        res += a + b;
        heap.push(a + b);
    }
    cout << res << endl;
    return 0;
}

ACW104.货仓选址

#include <iostream>
#include <algorithm>
using namespace std;

// 在数轴上选择一个点，使得该点到其它所有点的距离之和最小
// 解法：绝对值不等式即可
// 对于两个数，选点在两数之间即可选出最小值
// 对于n个数，第一个和最后一个数为一组，第二个和倒数第二个数一组
//      - 奇数选到中位数
//      - 偶数选到中间两数之间即可

const int N = 1e5 + 10;
int a[N];
long long res = 0;

int main() {
    int n;
    scanf("%d", &n);
    for (int i = 0; i < n; ++i) scanf("%d", &a[i]);
    sort(a, a + n);
    int idx = n / 2; // 可以不用判断奇数偶数，奇数时idx正好为中位数，偶数时为中间两数中右边一个数
    for (int i = 0; i < n; ++i) {
        res += abs(a[i] - a[idx]); 
    }
    printf("%d\n", res);
    return 0;
}

ACW25.耍杂技的牛

#include <iostream>
#include <algorithm>
#include <climits>
using namespace std;

const int N = 5e4 + 10;

// 任意两个相邻的牛之间满足（左边在上面，右边牛在下面) Wi + Si < Wi+1 + Si+1能够将最大值降到最低

struct Cow {
    int w, s;
    bool operator<(const Cow &c) {
        return w + s < c.w + c.s;
    }
} cow[N];
int n;

int main() {
    scanf("%d", &n);
    for (int i = 0; i < n; ++i) {
        scanf("%d%d", &cow[i].w, &cow[i].s);
    }

    sort(cow, cow + n);

    int weight = cow[0].w, maxDanger = 0 - cow[0].s;
    for (int i = 1; i < n; ++i) {
        maxDanger = max(maxDanger, weight - cow[i].s);
        weight += cow[i].w;
    }
    printf("%d\n", maxDanger);
    return 0;
}
