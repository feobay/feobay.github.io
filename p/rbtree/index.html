红黑树

很多STL底层就是通过红黑树实现，关联容器比如map

红黑树本质 -> 平衡二叉排序树

AVL树平衡条件——左右树高差别不超过1

SB_tree

五个条件

  1. 每个节点非黑即红
  2. 根节点是黑色
  3. 叶节点（NIL）是黑色
  4. 如果一个节点是红色，则它的两个子节点都是黑色的
  5. 从根节点出发到所有叶节点路径上，黑色节点数量相同

PS:

  1. 红黑树从根节点到叶子结点路径长最多差一倍

  2. 红黑树完全黑色符合定义

  3. 红黑树的约束比AVL树更加宽松

算法导论中描述的红黑树结构

插入红色更好调整

为什么？

根据条件每个路径上的黑色节点数相等，**插入黑色一定会失衡，插入红色不一定会失衡。**所以选择插入红色节点。

调整策略

  1. 插入调整站在祖父节点看
  2. 删除调整站在父节点看
  3. 插入和删除的情况处理一共五种

插入调整的情况

情况一：下方为某一颗子树

处理方法：回溯到15时，1和20修改为黑色，15修改为红色（红色上顶）

黑红红->红黑黑（保证红黑树的特性）

情况二：区别在于叔父节点是否为红色

处理方法：大右（左）旋，15改为黑色，20改为红色

其余情况均可以通过小左右旋达到该情况，类似于AVL的调整。

#include <stdio.h>
#include <stdlib.h>
#include <time.h>

typedef struct Node {
    int key;
    int color; // 0为红, 1为黑, 2为双红
    struct Node *lchild, *rchild;
} Node;

Node __NIL;
#define NIL (&__NIL)
__attribute__((constructor)) void init_NIL() {
    NIL->key = -1;
    NIL->color = 1;
    NIL->lchild = NIL->rchild = NIL;
}

Node *getNewNode(int key) {
    Node *p = (Node *)malloc(sizeof(Node));
    p->key = key;
    p->lchild = p->rchild = NIL;
    p->color = 0;
    return p;
}

inline bool has_red_child(Node *root) {
    return root->lchild->color == 0 || root->rchild->color == 0; // NIL节点避免了很多判断
}

Node *left_rotate(Node *root) {
    printf("left rotate : %d \n", root->key);
    Node *tmp = root->rchild;
    root->rchild = tmp->lchild;
    tmp->lchild = root;
    return tmp;
}

Node *right_rotate(Node *root) {
    printf("right rotate : %d \n", root->key);
    Node *tmp = root->lchild;
    root->lchild = tmp->rchild;
    tmp->rchild = root;
    return tmp;
}

const char *insert_maintain_str[] = {"", "LL", "LR", "RR", "RL"};

Node *insert_maintain(Node *root) { // 调整
    if (!has_red_child(root)) return root;
    if (root->lchild->color == 0 && root->rchild->color == 0) {
        // 情况一, 站在祖父节点角度看, 其两个孩子均为红色, 使红色上浮
        printf("Situation 1 : change color\n");
        root->color = 0;
        root->lchild->color = root->rchild->color = 1;
        return root;
    }
    if (root->lchild->color == 0 && !has_red_child(root->lchild)) return root;
    if (root->rchild->color == 0 && !has_red_child(root->rchild)) return root;
    // 下方失衡情况二, 站在祖父节点看, 它有一个孩子为红色
    int flag = 0;
    printf("Situation 2 : \n");
    if (root->lchild->color == 0) { // L为红色
        if (root->lchild->rchild->color == 0) { // LR
            root->lchild = left_rotate(root->lchild);
            flag = 2;
        } else {
            flag = 1;
        }
        root = right_rotate(root);
    } else { // R为红色
        if (root->rchild->lchild->color == 0) {
            root->rchild = right_rotate(root->rchild);
            flag = 3;
        } else {
            flag = 4;
        }
        root = left_rotate(root);
    }
    printf("Maitain Type : %s\n", insert_maintain_str[flag]);
    printf("Situation 2 : change color.\n");
    root->color = 0;
    root->lchild->color = root->rchild->color = 1;
    return root;
}

Node *__insert(Node *root, int key) {
    if (root == NIL) return getNewNode(key);
    if (root->key == key) return root;
    if (key < root->key) {
        root->lchild = __insert(root->lchild, key);
    } else {
        root->rchild = __insert(root->rchild, key);
    }
    return insert_maintain(root);
}

Node *insert(Node *root, int key) {
    root = __insert(root, key);
    root->color = 1; // 确保root为黑
    return root;
}

void clear(Node *root) {
    if (root == NIL) return ;
    clear(root->lchild);
    clear(root->rchild);
    free(root);
}

void print(Node *root) {
    printf("(%d | %d, %d, %d)\n",
            root->color, root->key,
            root->lchild->key,
            root->rchild->key);
}

void output(Node *root) {
    if (root == NIL) return ;
    print(root);
    output(root->lchild);
    output(root->rchild);
}

Node *rand_insert(Node *root) {
    int val = rand() % 100;
    root = insert(root, val);
    output(root);
    return root;
}

int main() {
    int n;
    srand(time(0));
    scanf("%d", &n);
    Node *root = NIL;
    for (int i = 0; i < n; ++i) {
        root = rand_insert(root);
    }
    clear(root);
    return 0;
}

删除调整

站在父节点看

删除度为0的，删除度为1的，删除度为2的

1. 删除的是一个红色节点

1.1 度为0，直接删

1.2 度为1，不可能（如果红色节点度为1，那么下面两条路径上的黑色节点数就不相等，所以不可能）

2. 删除的是一个黑色节点

2.1 度为1（那么其子节点必为红色，否则黑色节点数不平衡），直接删除当前节点，并将当前位置的黑色节点替换为其红色子节点，同时将该红色节点颜色为改为黑色

2.2 度为0

​ 若它有兄弟节点为黑色，那么直接删除本节点则会造成失衡。此时将其父节点标记为两重黑。

​ 没有兄弟节点表明它本身就是根节点

综上，要处理情况即为双重黑

实际删除调整策略

站在父节点向下看，x表示要删除的点

情况一 双重黑

要删除x点，将其父节点标记为双重黑

处理办法：brother调整为红色，x减少一重黑色，father增加一重黑色

情况二 RL

x为失衡节点，是左孩子，失衡节点的兄弟节点的左孩子为红色（RL)

将brother改为红色，再将其brother进行右旋，再将51改为黑色，转换为RR也就是情况三

情况三 RR类型

brother的右孩子为红色（RR），注意，brother的两个孩子都是红色时，仍然按照RR处理

对38进行左旋，因为不确定48的颜色，所以38需要置为黑色，同时为保证路径上黑色节点数相等，51改为38原来的颜色（此处为红色），72更改为黑色

#include <time.h>

    int color; // 0为红, 1为黑, 2为双红

} Node;

Node __NIL;
#define NIL (&__NIL)
__attribute__((constructor)) void init_NIL() {

    NIL->color = 1;
    NIL->lchild = NIL->rchild = NIL;
}

Node *getNewNode(int key) {

}

inline bool has_red_child(Node *root) {

Node *left_rotate(Node *root) {

const char *insert_maintain_str[] = {"", "LL", "LR", "RR", "RL"};

    if (!has_red_child(root)) return root;

        // 情况一, 站在祖父节点角度看, 其两个孩子均为红色, 使红色上浮

const char *maintain_str[] = {"", "LL", "LR", "RR", "RL"};

Node *erase_maintain(Node *root) {
    if (root->lchild->color != 2 && root->rchild->color != 2) return root;
    if (has_red_child(root)) {
        printf("Situation X : brother is red!\n");
        int flag = 0;
        root->color = 0; // 旋转前根置为红
        if (root->lchild->color == 0) { 
            root = right_rotate(root);
            flag = 0; // 如果大右旋，需要继续进入右子树进行调整
        } else {
            root = left_rotate(root);
            flag = 1; // 反之，进入左子树
        }
        root->color = 1; // 旋转后，新根置为黑
        if (flag == 0) root->rchild = erase_maintain(root->rchild);
        else root->lchild = erase_maintain(root->lchild);
        return root;
    }
    // 情况一 两个孩子为黑（双重黑），且兄弟的孩子不为红
    if ((root->lchild->color == 2 && !has_red_child(root->rchild)) ||
        (root->rchild->color == 2 && !has_red_child(root->lchild))) {
        printf("Situation 1 : change color done!\n");
        root->color += 1; // 根节点颜色加上一重黑，两个孩子减去一重黑
        root->lchild->color -= 1;
        root->rchild->color -= 1;
        return root;
    }
    
    int maintain_type = 0;

    if (root->rchild->color == 2) { //L
        root->rchild->color = 1; // 先将双重黑置为一重黑
        if (root->lchild->lchild->color != 0) {//LR
            root->lchild->color = 0; //左子树根节点置为红
            root->lchild = left_rotate(root->lchild); //小左旋
            root->lchild->color = 1; //左旋后，左子树根节点从红置为黑
            maintain_type = 2;
        } else {
            maintain_type = 1;
        }
        root = right_rotate(root); // 大右旋
        root->color = root->rchild->color; // 大右旋之后，根节点颜色改为右孩子的颜色
    } else { //根据上面的情况镜像操作
        root->lchild->color = 1;
        if (root->rchild->color != 0) { //RL
            root->rchild->color = 0;
            root->rchild = right_rotate(root->rchild);
            root->rchild->color = 1;
            maintain_type = 3;
        } else {
            maintain_type = 4;
        }
        root = left_rotate(root);
        root->color = root->lchild->color;
    }
    // 最后将左孩子和右孩子的颜色置为黑
    root->lchild->color = root->rchild->color = 1;
    printf("Maintain type : %s\n", maintain_str[maintain_type]);
    return root;
}

Node *predecessor(Node *root) {
    Node *l = root->lchild;
    while (l->rchild != NIL) {
        l = l->rchild;
    } 
    return l;
}

Node *__erase(Node *root, int key) {
    if (root == NIL) return NIL;
    if (key < root->key) {
        root->lchild = __erase(root->lchild, key);
    } else if (key > root->key) {
        root->rchild = __erase(root->rchild, key);
    } else {
        if (root->lchild == NIL || root->rchild == NIL) { // 判断出度是否为1，出度为1只可能是黑色根，红色子
            Node *temp = root->lchild == NIL ? root->lchild : root->rchild;
            temp->color += root->color; // 删除根，子节点红变黑
            free(root);
            return temp;
        } else { // 度为0或2
            Node *temp = predecessor(root); // 先找前驱，待删除的度为2
            root->key = temp->key;
            root->lchild = __erase(root->lchild, temp->key);
        }
    }
    return erase_maintain(root);
}

Node *erase(Node *root, int key) {
    root = __erase(root, key);
    root->color = 1;
    return root;
}

void clear(Node *root) {
    if (root == NIL) return ;
    clear(root->lchild);
    clear(root->rchild);
    free(root);
}

void print(Node *root) {
    printf("(%d | %d, %d, %d)\n",
            root->color, root->key,
            root->lchild->key,
            root->rchild->key);
}

void __output(Node *root) {
    if (root == NIL) return;
    print(root);
    __output(root->lchild);
    __output(root->rchild);
}

void output(Node *root) {
    __output(root);
    printf("NIL : ");
    print(NIL);
    return ;
}

Node *rand_insert(Node *root) {
    int val = rand() % 100;
    root = insert(root, val);
    output(root);
    return root;
}

int main() {
    int n;
    srand(time(0));
    scanf("%d", &n);
    Node *root = NIL;
    for (int i = 0; i < n; ++i) {
        root = rand_insert(root);
    }
    while (~scanf("%d", &n)) {
        root = erase(root, n);
    }
    clear(root);
    return 0;
}
